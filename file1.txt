#include <iostream>

using namespace std;

int main()
{
    int number;
    cout << "Enter the number of height= ";
    cin >> number;
    int height[number];
    cout << "Enter the values of height in meters= ";
    for (int i; i<number; i++){
        cin >> height[i];
    }
    int i=0;
    while( i<number){
        if (height[i]>=height[i+1]){
           height[i+1]=height[i];
        }
        else if (height[i]<height[i+1]){
           height[i+1]=height[i+1];
        }
        i++;
    }
    cout << height[number-1];

    return 0;
}
